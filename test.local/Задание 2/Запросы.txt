1. SELECT hosp.name as name, count(DISTINCT service.kod_patient) as MAX FROM hosp LEFT JOIN service ON hosp.kod = service.kod_hosp GROUP BY service.kod_hosp HAVING max>10
2. SELECT patient.name as name, count(service.kod_hosp) as MAX FROM patient LEFT JOIN service ON patient.kod = service.kod_patient GROUP BY service.kod_patient HAVING max>2
3. SELECT hosp.name as name, count(*) as MAX FROM hosp LEFT JOIN service ON hosp.kod = service.kod_hosp GROUP BY service.kod_hosp HAVING max>10 ORDER BY max DESC LIMIT 1
